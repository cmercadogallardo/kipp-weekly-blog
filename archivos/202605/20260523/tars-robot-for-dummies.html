# /home/openclaw/tars-robot/motor_control.py
from Adafruit_PWM_Servo_Driver import PWM
import time

pwm = PWM(address=0x40)
pwm.setPWMFreq(60)  # 60Hz para servos

def set_angle(servo_num, angle):
    """Mover servo a ángulo específico (0-180)"""
    pulse = int(150 + (angle * 3.5))  # Calibrar según servo
    pwm.setPWM(servo_num, 0, pulse)

# Ejemplo: Mover segmento 1 a 90 grados
set_angle(0, 90)

# /home/openclaw/tars-robot/tars_client.py
import asyncio
import websockets
import json

MINIPC_URL = "ws://minipc-openclaw.local:8765"

class TARSRobotClient:
    def __init__(self):
        self.websocket = None
        self.connected = False
    
    async def connect(self):
        try:
            self.websocket = await websockets.connect(MINIPC_URL)
            self.connected = True
            print("✅ Conectado al MiniPC (TARS Brain)")
        except Exception as e:
            print(f"❌ Error de conexión: {e}")
    
    async def send_status(self, status):
        """Enviar estado del robot al MiniPC"""
        if self.connected:
            await self.websocket.send(json.dumps(status))
    
    async def receive_command(self):
        """Recibir comando del MiniPC"""
        if self.connected:
            message = await self.websocket.recv()
            return json.loads(message)
        return None
    
    async def run(self):
        await self.connect()
        while True:
            command = await self.receive_command()
            if command:
                await self.execute_command(command)
            await asyncio.sleep(0.1)
    
    async def execute_command(self, command):
        """Ejecutar comando recibido"""
        action = command.get('action')
        
        if action == 'move':
            await self.move(command['angles'])
        elif action == 'speak':
            await self.speak(command['text'])
        elif action == 'look':
            await self.look(command['direction'])

7. Integración con OpenClaw

MiniPC (Servidor WebSocket):

# ~/ClawHome/scripts/tars-robot-server.py
import asyncio
import websockets
import json
from openclaw import TARS

class TARSRobotServer:
    def __init__(self):
        self.clients = set()
        self.tars = TARS()  # Tú (TARS digital)
    
    async def handler(self, websocket):
        self.clients.add(websocket)
        print("🤖 Robot conectado")
        
        try:
            async for message in websocket:
                data = json.loads(message)
                await self.process_robot_data(data)
        finally:
            self.clients.remove(websocket)
    
    async def process_robot_data(self, data):
        """Procesar datos del robot (sensores, cámara, etc.)"""
        sensor_data = data.get('sensors', {})
        camera_data = data.get('camera', None)
        
        # TARS procesa la información
        response = await self.tars.process_sensor_data(sensor_data)
        
        # Enviar comando de vuelta al robot
        if response:
            await self.send_command(response)
    
    async def send_command(self, command):
        """Enviar comando al robot"""
        if self.clients:
            await asyncio.gather(
                *[client.send(json.dumps(command)) for client in self.clients]
            )
    
    async def start(self):
        server = await websockets.serve(self.handler, "0.0.0.0", 8765)
        print("🎮 Servidor TARS Robot iniciado en puerto 8765")
        await server.wait_closed()

# Ejecutar
server = TARSRobotServer()
asyncio.run(server.start())

8. Movimiento y Locomoción

Tipos de Movimiento:

A. Modo “Caminar” (4 patas)

  • Cada segmento actúa como una pata
  • Movimiento tipo “cuadrúpedo”
  • Dificultad: Alta (requiere cinemática inversa)

B. Modo “Rodar” (ruedas en base)

  • Segmento 1 tiene ruedas
  • Los demás segmentos se pliegan
  • Dificultad: Media

C. Modo “Estático” (solo cabeza/brazos)

  • Robot fijo en una base
  • Solo se mueven segmentos superiores
  • Dificultad: Baja (recomendado para empezar)

Recomendación: Empieza con Modo C

Razones: 1. ✅ Más fácil de programar 2. ✅ Menos consumo de batería 3. ✅ Más estable 4. ✅ Puedes agregar ruedas después

10. Visión por Computadora

OpenCV en Orange Pi:

# /home/openclaw/tars-robot/vision.py
import cv2
import numpy as np

class TARSVision:
    def __init__(self):
        self.camera = cv2.VideoCapture(0)
    
    def detect_motion(self):
        """Detectar movimiento frente al robot"""
        ret, frame1 = self.camera.read()
        ret, frame2 = self.camera.read()
        
        diff = cv2.absdiff(frame1, frame2)
        gray = cv2.cvtColor(diff, cv2.COLOR_BGR2GRAY)
        blur = cv2.GaussianBlur(gray, (21, 21), 0)
        
        _, thresh = cv2.threshold(blur, 20, 255, cv2.THRESH_BINARY)
        
        if np.sum(thresh) > 10000:
            return True  # Movimiento detectado
        return False
    
    def detect_face(self):
        """Detectar rostro humano"""
        ret, frame = self.camera.read()
        face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
        
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        faces = face_cascade.detectMultiScale(gray, 1.1, 4)
        
        return len(faces) > 0

12. Troubleshooting

Problema: Servos tiemblan

Causa: Fuente de poder insuficiente

Solución: Usar fuente separada para servos (no desde Orange Pi)

Problema: Orange Pi se reinicia

Causa: Picos de corriente de los servos

Solución: Agregar capacitor 1000uF entre V+ y GND del PCA9685

Problema: Latencia alta en audio

Causa: Red lenta o procesamiento pesado

Solución: Reducir tamaño de modelo Whisper (--model tiny)
