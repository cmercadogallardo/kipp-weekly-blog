Resumen Ejecutivo

Investigación sobre cómo construir una librería para conectar Firebase Firestore y escuchar cambios en una colección desde un servidor local, permitiendo ejecutar hooks cuando se detectan modificaciones.

Estrategias Identificadas

Opción A: SDK de Firebase + Realtime Listener (⭐ RECOMENDADA)

Conexión persistente en tiempo real usando onSnapshot.

Instalación:

npm install firebase

Implementación base:

// firestore-listener.js
const { initializeApp } = require('firebase/app');
const { getFirestore, collection, onSnapshot, query, where, orderBy } = require('firebase/firestore');

// Configuración
const firebaseConfig = {
  apiKey: process.env.FIREBASE_API_KEY,
  authDomain: process.env.FIREBASE_AUTH_DOMAIN,
  projectId: process.env.FIREBASE_PROJECT_ID,
  storageBucket: process.env.FIREBASE_STORAGE_BUCKET,
  messagingSenderId: process.env.FIREBASE_MESSAGING_SENDER_ID,
  appId: process.env.FIREBASE_APP_ID
};

// Inicializar
const app = initializeApp(firebaseConfig);
const db = getFirestore(app);

// Listener en tiempo real
const unsubscribe = onSnapshot(
  query(
    collection(db, 'comandos'),
    orderBy('timestamp', 'desc'),
    where('estado', 'in', ['pendiente', 'procesando'])
  ),
  (snapshot) => {
    snapshot.docChanges().forEach((change) => {
      if (change.type === 'added' || change.type === 'modified') {
        const data = change.doc.data();
        console.log(`[${change.type}] Comando recibido:`, data);
        
        // Ejecutar hook local
        ejecutarHook(data);
        
        // Marcar como procesado
        change.doc.ref.update({ estado: 'procesado', procesadoEn: new Date() });
      }
    });
  },
  (error) => {
    console.error('Error en listener:', error);
  }
);

// Función para ejecutar hooks
async function ejecutarHook(comando) {
  try {
    const { tipo, payload, origen } = comando;
    
    switch (tipo) {
      case 'exec':
        const { exec } = require('child_process');
        exec(payload.command, (error, stdout, stderr) => {
          console.log('Resultado:', stdout || stderr);
        });
        break;
        
      case 'webhook':
        const fetch = require('node-fetch');
        await fetch(payload.url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload.data)
        });
        break;
        
      case 'script':
        const scriptPath = `./hooks/${payload.script}.js`;
        const hook = require(scriptPath);
        await hook.run(payload.params);
        break;
        
      default:
        console.warn(`Tipo de comando desconocido: ${tipo}`);
    }
  } catch (error) {
    console.error('Error ejecutando hook:', error);
  }
}

// Mantener proceso activo
process.on('SIGINT', () => {
  console.log('Cerrando listener...');
  unsubscribe();
  process.exit(0);
});

console.log('🔥 Firestore listener activo. Escuchando colección "comandos"...');

Ventajas: - ✅ Conexión persistente en tiempo real - ✅ Bajo consumo de recursos - ✅ Reconección automática ante fallos - ✅ Filtrado del lado del servidor - ✅ Latencia mínima (<1s)

Opción B: Cloud Functions + Webhook HTTP

Firebase Cloud Functions que dispara webhook a servidor local.

Instalación:

npm install -g firebase-tools
firebase init functions

Cloud Function:

// functions/index.js
const functions = require('firebase-functions');
const admin = require('firebase-admin');
const fetch = require('node-fetch');

admin.initializeApp();
const db = admin.firestore();

exports.onComandoCreado = functions.firestore
  .document('comandos/{docId}')
  .onCreate(async (snap, context) => {
    const data = snap.data();
    const LOCAL_SERVER_URL = process.env.LOCAL_SERVER_URL || 'http://tu-ip-publico:3000';
    
    try {
      await fetch(`${LOCAL_SERVER_URL}/hooks/firebase`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${process.env.WEBHOOK_SECRET}`
        },
        body: JSON.stringify({
          id: context.params.docId,
          ...data,
          timestamp: new Date().toISOString()
        })
      });
      
      // Marcar como enviado
      await snap.ref.update({ enviado: true, enviadoEn: admin.firestore.FieldValue.serverTimestamp() });
    } catch (error) {
      console.error('Error enviando webhook:', error);
      await snap.ref.update({ error: error.message });
    }
  });

exports.onComandoModificado = functions.firestore
  .document('comandos/{docId}')
  .onUpdate(async (change, context) => {
    const before = change.before.data();
    const after = change.after.data();
    
    // Solo triggerar si cambió el estado a 'pendiente'
    if (before.estado !== 'pendiente' && after.estado === 'pendiente') {
      // Mismo lógica de webhook
    }
  });

Servidor local receptor:

// server.js
const express = require('express');
const app = express();

app.use(express.json());

app.post('/hooks/firebase', (req, res) => {
  const { authorization } = req.headers;
  
  // Validar token
  if (authorization !== `Bearer ${process.env.WEBHOOK_SECRET}`) {
    return res.status(401).json({ error: 'No autorizado' });
  }
  
  const comando = req.body;
  console.log('Comando recibido:', comando);
  
  // Ejecutar hook
  ejecutarHook(comando);
  
  res.json({ received: true });
});

app.listen(3000, () => {
  console.log('Servidor escuchando en puerto 3000');
});

Requisitos: - ⚠️ Servidor local debe ser accesible públicamente - ⚠️ Usar ngrok, Cloudflare Tunnel, o IP pública estática - ⚠️ Mayor latencia (2-5s) - ✅ Útil si ya usas Cloud Functions para otras cosas

Costos: - Cloud Functions: 2M invocaciones/mes gratis, luego $0.40/100K - Networking: Gratis dentro de límites

Opción C: Polling con Admin SDK (Fallback)

Polling periódico usando Firebase Admin SDK.

Instalación:

npm install firebase-admin

Implementación:

// polling-listener.js
const admin = require('firebase-admin');
const serviceAccount = require('./service-account.json');

admin.initializeApp({
  credential: admin.credential.cert(serviceAccount)
});

const db = admin.firestore();
let lastTimestamp = new Date();

async function pollFirestore() {
  try {
    const snapshot = await db.collection('comandos')
      .where('timestamp', '>', lastTimestamp)
      .where('estado', 'in', ['pendiente', 'reintentar'])
      .orderBy('timestamp', 'asc')
      .limit(100)
      .get();
    
    if (snapshot.empty) {
      console.log('Sin cambios nuevos');
      return;
    }
    
    console.log(`${snapshot.size} comando(s) nuevo(s) detectado(s)`);
    
    const promises = [];
    snapshot.forEach((doc) => {
      const data = doc.data();
      promises.push(
        ejecutarHook(data).then(() => {
          return doc.ref.update({
            estado: 'procesado',
            procesadoEn: admin.firestore.FieldValue.serverTimestamp()
          });
        })
      );
      
      // Actualizar último timestamp
      if (data.timestamp > lastTimestamp) {
        lastTimestamp = data.timestamp;
      }
    });
    
    await Promise.all(promises);
  } catch (error) {
    console.error('Error en polling:', error);
  }
}

// Ejecutar cada 5 segundos
setInterval(pollFirestore, 5000);

// Ejecutar inmediatamente al inicio
pollFirestore();

console.log('🔄 Polling iniciado (intervalo: 5s)');

Ventajas: - ✅ Simple de implementar - ✅ No requiere conexión persistente - ✅ Funciona detrás de firewall sin configuración especial

Desventajas: - ⚠️ Latencia de hasta el intervalo de polling - ⚠️ Consumo de lecturas de Firestore (costos) - ⚠️ No es “tiempo real” verdadero

Costos estimados: - Polling cada 5s = 12 lecturas/min = 17,280 lecturas/día - Firestore: 50K lecturas/día gratis, luego $0.06/100K

src/listener.js

const { getFirestore, collection, onSnapshot, query, where, orderBy } = require('firebase/firestore');
const config = require('./config');
const logger = require('./utils/logger');
const hooks = require('./hooks');

class FirestoreListener {
  constructor(db) {
    this.db = db;
    this.unsubscribe = null;
  }

  start() {
    const { collectionName } = config.listener;
    
    logger.info(`Iniciando listener en colección "${collectionName}"...`);
    
    this.unsubscribe = onSnapshot(
      query(
        collection(this.db, collectionName),
        orderBy('timestamp', 'desc'),
        where('estado', 'in', ['pendiente', 'reintentar'])
      ),
      (snapshot) => {
        snapshot.docChanges().forEach(async (change) => {
          if (change.type === 'added' || change.type === 'modified') {
            const data = change.doc.data();
            const docRef = change.doc.ref;
            
            logger.info(`[${change.type}] Comando recibido:`, {
              id: change.doc.id,
              tipo: data.tipo,
              origen: data.origen
            });
            
            try {
              // Validar origen
              if (!this.validateOrigin(data.origen)) {
                logger.warn(`Origen no autorizado: ${data.origen}`);
                return;
              }
              
              // Ejecutar hook
              await hooks.execute(data);
              
              // Marcar como procesado
              await docRef.update({
                estado: 'procesado',
                procesadoEn: new Date()
              });
              
              logger.info(`Comando procesado: ${change.doc.id}`);
            } catch (error) {
              logger.error(`Error procesando comando:`, error);
              await docRef.update({
                estado: 'error',
                error: error.message,
                reintentos: (data.reintentos || 0) + 1
              });
            }
          }
        });
      },
      (error) => {
        logger.error('Error en listener:', error);
        // Reconección automática manejada por Firebase SDK
      }
    );
  }

  stop() {
    if (this.unsubscribe) {
      this.unsubscribe();
      logger.info('Listener detenido');
    }
  }

  validateOrigin(origen) {
    const { allowedOrigins } = config.security;
    if (allowedOrigins.length === 0) return true; // Sin restricciones
    return allowedOrigins.includes(origen);
  }
}

module.exports = FirestoreListener;

src/hooks/index.js

const { exec } = require('child_process');
const fetch = require('node-fetch');
const fs = require('fs');
const path = require('path');
const logger = require('../utils/logger');

class HookExecutor {
  async execute(comando) {
    const { tipo, payload } = comando;
    
    switch (tipo) {
      case 'exec':
        return this.executeShell(payload);
      
      case 'webhook':
        return this.executeWebhook(payload);
      
      case 'script':
        return this.executeScript(payload);
      
      default:
        throw new Error(`Tipo de comando desconocido: ${tipo}`);
    }
  }

  async executeShell(payload) {
    return new Promise((resolve, reject) => {
      exec(payload.command, { timeout: payload.timeout || 30000 }, (error, stdout, stderr) => {
        if (error) {
          reject(error);
          return;
        }
        logger.info('Shell output:', stdout || stderr);
        resolve({ stdout, stderr });
      });
    });
  }

  async executeWebhook(payload) {
    const response = await fetch(payload.url, {
      method: payload.method || 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...payload.headers
      },
      body: JSON.stringify(payload.data),
      timeout: payload.timeout || 10000
    });
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }
    
    return response.json();
  }

  async executeScript(payload) {
    const scriptPath = path.join(__dirname, payload.script);
    
    if (!fs.existsSync(scriptPath)) {
      throw new Error(`Script no encontrado: ${scriptPath}`);
    }
    
    const hook = require(scriptPath);
    
    if (typeof hook.run !== 'function') {
      throw new Error('Script debe exportar función run()');
    }
    
    return hook.run(payload.params);
  }
}

module.exports = new HookExecutor();

src/index.js

const { initializeApp } = require('firebase/app');
const { getFirestore } = require('firebase/firestore');
const config = require('./config');
const FirestoreListener = require('./listener');
const logger = require('./utils/logger');

async function main() {
  logger.info('🚀 Iniciando Firestore Listener...');
  
  // Inicializar Firebase
  const app = initializeApp(config.firebase);
  const db = getFirestore(app);
  
  // Crear y iniciar listener
  const listener = new FirestoreListener(db);
  listener.start();
  
  // Manejar señales de cierre
  process.on('SIGINT', () => {
    logger.info('Recibida señal SIGINT, cerrando...');
    listener.stop();
    process.exit(0);
  });
  
  process.on('SIGTERM', () => {
    logger.info('Recibida señal SIGTERM, cerrando...');
    listener.stop();
    process.exit(0);
  });
  
  logger.info('✅ Listener activo. Presiona Ctrl+C para detener.');
}

main().catch((error) => {
  logger.error('Error fatal:', error);
  process.exit(1);
});

Consideraciones de Seguridad

🔐 Autenticación

  1. Service Account - Usar cuenta de servicio con permisos mínimos necesarios
  2. Reglas de Firestore - Restringir lectura/escritura:
// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /comandos/{docId} {
      allow write: if request.auth != null && request.auth.token.email == 'service@your-project.iam.gserviceaccount.com';
      allow read: if request.auth != null && request.auth.token.email == 'service@your-project.iam.gserviceaccount.com';
    }
  }
}
  1. Validación de origen - Campo origen en documento debe estar en whitelist
  2. Tokens locales - Validar autenticidad antes de ejecutar comandos

🛡️ Autorización

  • Limitar tipos de comandos permitidos
  • Timeout máximo para ejecución de shells
  • No ejecutar comandos con pipes, redirects, o subshells peligrosos
